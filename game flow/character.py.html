import pygame
import game
from button import Button, ImageButton

window = pygame.display.set_mode((game.screen_width, game.screen_height))

def draw_character():
    window.fill(game.WHITE)

class Character:
    def __init__(self, screen):
        self.screen = screen
        self.dolls = [pygame.image.load('images/barbie1.png'), pygame.image.load('images/barbie2.png'),
                      pygame.image.load('images/barbie3.png'), pygame.image.load('images/barbie4.png')]
        self.currImage = 0
        self.font = pygame.font.SysFont('arial', int(game.screen_width/25))

        self.width = game.screen_width
        self.height = game.screen_height
        arrow_size = self.width/8


        self.buttons = [ImageButton('images/arrow.png', self.width/8,
                                    (self.height-arrow_size)/2, True, arrow_size),
                        ImageButton('images/arrow.png', 7*self.width/8 - arrow_size,
                                    (self.height-arrow_size)/2, False, arrow_size)]
        self.select = Button("select", (3*self.width/4)/2, 4*self.height/5, self.width/4, self.height/12, game.PINK, game.LIGHT_PINK)
        self.play = False

    def draw(self):
        self.screen.fill(game.WHITE)
        if not self.play:
            scale = 0.5
            iwidth = int(self.width*scale)
            iheight = int(self.width*scale)
            ix = (self.width - iwidth)/2
            iy = (self.height - iheight)/2
            self.screen.blit(pygame.transform.scale(self.dolls[self.currImage], (iwidth, iheight)), (ix, iy))

        for button in self.buttons:
            button.draw(self.screen)

        self.select.draw(self.screen, self.font)


    def state(self, event):
        if event.type == pygame.MOUSEBUTTONDOWN and not self.play:
            if self.select.clicked(pygame.mouse.get_pos()):
                self.play = True
            elif (self.buttons[1].clicked(pygame.mouse.get_pos())):
                if self.currImage != 3:
                    self.currImage += 1
            elif self.buttons[0].clicked(pygame.mouse.get_pos()):
                if self.currImage != 0:
                    self.currImage -=1
        if self.play:
            self.play = False
            return "play"

        self.select.collide(pygame.mouse.get_pos())

        return "character"
